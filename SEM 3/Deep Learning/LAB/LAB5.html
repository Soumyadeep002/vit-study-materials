def sigmoid(z):
    return 1 / (1 + np.exp(-z))

def sigmoid_derivative(z):
    s = sigmoid(z)
    return s * (1 - s)

def mse(y_true, y_pred):
    return np.mean((y_true - y_pred) ** 2)

def mse_derivative(y_true, y_pred):
    return (y_pred - y_true)

# CLASSICAL MOMENTUM TRAINING 
print("\nTraining with Momentum")
for epoch in range(epochs):
    total_loss = 0
    
    for i in range(len(X)):
        xi = X[i].reshape(1, -1)   
        yi = y[i]

        z = np.dot(xi, w) + b
        y_pred = sigmoid(z)

        loss = mse(yi, y_pred)
        total_loss += loss

        dL_dy = mse_derivative(yi, y_pred) 
        dy_dz = sigmoid_derivative(z)      

        dL_dz = dL_dy * dy_dz            
        dL_dw = xi.T * dL_dz            
        dL_db = dL_dz.item()            

        v_w = momentum * v_w - lr * dL_dw
        v_b = momentum * v_b - lr * dL_db

        w += v_w
        b += v_b

    if epoch % 20 == 0:
        print(f"Epoch {epoch}, Loss={total_loss/len(X):.4f}")

    total_loss = 0
    
    for i in range(len(X)):
        xi = X[i].reshape(1, -1)
        yi = y[i]

        w_lookahead = w + momentum * v_w
        b_lookahead = b + momentum * v_b

        z = np.dot(xi, w_lookahead) + b_lookahead
        y_pred = sigmoid(z)

        loss = mse(yi, y_pred)
        total_loss += loss

        dL_dy = mse_derivative(yi, y_pred)
        dy_dz = sigmoid_derivative(z)

        dL_dz = dL_dy * dy_dz
        dL_dw = xi.T * dL_dz
        dL_db = dL_dz.item()

        v_w = momentum * v_w - lr * dL_dw
        v_b = momentum * v_b - lr * dL_db

        w += v_w
        b += v_b
